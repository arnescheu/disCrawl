CREATE TABLE summary (
	pdb_id VARCHAR(10) NOT NULL, 
	abort BOOLEAN, 
	pass_filesize BOOLEAN, 
	filesize DECIMAL, 
	pass_structure BOOLEAN, 
	pass_dictionary BOOLEAN, 
	pass_analysis BOOLEAN, 
	model_count DECIMAL, 
	distances DECIMAL, 
	inter_count DECIMAL, 
	dist_count DECIMAL, 
	hit_count DECIMAL, 
	cys_count DECIMAL, 
	name VARCHAR(250), 
	head VARCHAR(250), 
	genes VARCHAR(250), 
	genus VARCHAR(200), 
	host VARCHAR(200), 
	method VARCHAR(30), 
	bio INTEGER, 
	bio_list VARCHAR(30), 
	bio_count DECIMAL, 
	ref_ids VARCHAR(30), 
	poly_ids VARCHAR(30), 
	deposition VARCHAR(10), 
	resolution VARCHAR(10), 
	distance DECIMAL, 
	"d1_X_C" DECIMAL, 
	"d2_X_CA" DECIMAL, 
	"d3_X_N" DECIMAL, 
	"d4_CA_C" DECIMAL, 
	"d5_CA_CA" DECIMAL, 
	"d6_CA_N" DECIMAL, 
	"d7_N_C" DECIMAL, 
	"d8_N_CA" DECIMAL, 
	"d9_N_N" DECIMAL, 
	model VARCHAR(3), 
	chain_count DECIMAL, 
	dynamic_chain VARCHAR(3), 
	dynamic_chain_poly_id VARCHAR(3), 
	dynamic_chain_res_count INTEGER, 
	dynamic_resn VARCHAR(3), 
	dynamic_resi INTEGER, 
	dynamic_pos INTEGER, 
	dynamic_name VARCHAR(2), 
	x_name VARCHAR(2), 
	nterm BOOLEAN, 
	dynamic_order BOOLEAN, 
	static_chain VARCHAR(3), 
	static_chain_poly_id VARCHAR(3), 
	static_chain_res_count INTEGER, 
	static_resn VARCHAR(3), 
	static_resi INTEGER, 
	static_pos INTEGER, 
	static_name VARCHAR(2), 
	static_order BOOLEAN, 
	chain_poly_id_eq BOOLEAN, 
	inter BOOLEAN, 
	job_id VARCHAR(30), 
	timestamp VARCHAR(30), 
	PRIMARY KEY (pdb_id), 
	CHECK (abort IN (0, 1)), 
	CHECK (pass_filesize IN (0, 1)), 
	CHECK (pass_structure IN (0, 1)), 
	CHECK (pass_dictionary IN (0, 1)), 
	CHECK (pass_analysis IN (0, 1)), 
	CHECK (nterm IN (0, 1)), 
	CHECK (dynamic_order IN (0, 1)), 
	CHECK (static_order IN (0, 1)), 
	CHECK (chain_poly_id_eq IN (0, 1)), 
	CHECK (inter IN (0, 1))
)
